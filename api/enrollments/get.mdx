---
title: "Get Enrollment"
api: "GET /api/enrollments/{enrollment_id}"
description: "Retrieve detailed information about a specific enrollment"
---

## Path Parameters

<ParamField path="enrollment_id" type="uuid" required>
  Unique identifier of the enrollment
</ParamField>

## Response

Returns the complete enrollment object with all associated documents.

<ResponseField name="id" type="uuid">
  Unique enrollment identifier
</ResponseField>

<ResponseField name="school_id" type="uuid">
  School identifier
</ResponseField>

<ResponseField name="student_id" type="uuid">
  Student identifier
</ResponseField>

<ResponseField name="school_year_id" type="uuid">
  School year identifier
</ResponseField>

<ResponseField name="grade_level" type="string">
  Grade level (e.g., "1°", "2°", "10°", "11°")
</ResponseField>

<ResponseField name="group_code" type="string" nullable>
  Group or class code (e.g., "A", "B", "1A")
</ResponseField>

<ResponseField name="shift" type="string" nullable>
  School shift (e.g., "morning", "afternoon")
</ResponseField>

<ResponseField name="status" type="string">
  Current enrollment status
</ResponseField>

<ResponseField name="status_history" type="array">
  Array of status changes with timestamp and user
  
  <Expandable title="Status History Item">
    <ResponseField name="status" type="string">
      The status at this point in time
    </ResponseField>
    
    <ResponseField name="changed_at" type="datetime">
      When the status was changed
    </ResponseField>
    
    <ResponseField name="changed_by" type="uuid">
      User ID who made the change
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="simat_status" type="object">
  SIMAT (Colombian education system) status data
</ResponseField>

<ResponseField name="extra_data" type="object">
  Additional custom data for the enrollment
</ResponseField>

<ResponseField name="student_name" type="string">
  Full name of the student
</ResponseField>

<ResponseField name="school_year_name" type="string">
  Name of the school year
</ResponseField>

<ResponseField name="documents" type="array">
  Array of enrollment document objects
  
  <Expandable title="Document Object">
    <ResponseField name="id" type="uuid">
      Document identifier
    </ResponseField>
    
    <ResponseField name="school_id" type="uuid">
      School identifier
    </ResponseField>
    
    <ResponseField name="enrollment_id" type="uuid">
      Enrollment identifier
    </ResponseField>
    
    <ResponseField name="document_type" type="string">
      Type of document (e.g., "birth_certificate", "id_card")
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Document status (e.g., "uploaded", "validated")
    </ResponseField>
    
    <ResponseField name="r2_object_key" type="string">
      Storage key for the document file
    </ResponseField>
    
    <ResponseField name="file_name" type="string">
      Original filename of the uploaded document
    </ResponseField>
    
    <ResponseField name="uploaded_at" type="datetime">
      When the document was uploaded
    </ResponseField>
    
    <ResponseField name="validated_at" type="datetime" nullable>
      When the document was validated (if applicable)
    </ResponseField>
    
    <ResponseField name="validated_by" type="uuid" nullable>
      User ID who validated the document
    </ResponseField>
    
    <ResponseField name="accepted_by_guardian_id" type="uuid" nullable>
      Guardian ID who accepted the document
    </ResponseField>
    
    <ResponseField name="document_version" type="string" nullable>
      Version of the document
    </ResponseField>
    
    <ResponseField name="metadata_json" type="object">
      Additional metadata for the document
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created_at" type="datetime">
  Timestamp when the enrollment was created
</ResponseField>

<ResponseField name="updated_at" type="datetime">
  Timestamp when the enrollment was last updated
</ResponseField>

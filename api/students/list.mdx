---
title: 'List Students'
openapi: 'GET /students'
---

Retrieves a paginated list of students enrolled in the school.

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination (minimum: 1)
</ParamField>

<ParamField query="page_size" type="integer" default="20">
  Number of items per page (minimum: 1, maximum: 100)
</ParamField>

<ParamField query="search" type="string" optional>
  Search term to filter students by full name or document number (case-insensitive)
</ParamField>

<ParamField query="grade" type="string" optional>
  Filter students by grade level (partial match)
</ParamField>

<ParamField query="is_active" type="boolean" default="true">
  Filter by active status. Set to `null` to retrieve all students regardless of status
</ParamField>

## Response

Returns a paginated list of students with their associated guardians.

<ResponseField name="total" type="integer">
  Total number of students matching the filters
</ResponseField>

<ResponseField name="page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="page_size" type="integer">
  Number of items per page
</ResponseField>

<ResponseField name="items" type="array">
  Array of student objects
  
  <Expandable title="Student Object">
    <ResponseField name="id" type="uuid">
      Unique identifier for the student
    </ResponseField>
    
    <ResponseField name="school_id" type="uuid">
      School identifier
    </ResponseField>
    
    <ResponseField name="user_id" type="uuid" optional>
      Associated user account ID (if linked)
    </ResponseField>
    
    <ResponseField name="document_type" type="string">
      Type of identification document. Valid values: `TI`, `RC`, `CC`, `CE`, `NUIP`, `PASSPORT`
    </ResponseField>
    
    <ResponseField name="document_number" type="string">
      Identification document number
    </ResponseField>
    
    <ResponseField name="full_name" type="string">
      Student's full name
    </ResponseField>
    
    <ResponseField name="birth_date" type="date" optional>
      Date of birth in YYYY-MM-DD format
    </ResponseField>
    
    <ResponseField name="gender" type="string" optional>
      Student's gender
    </ResponseField>
    
    <ResponseField name="is_active" type="boolean">
      Whether the student is currently active
    </ResponseField>
    
    <ResponseField name="extra_data" type="object">
      Additional custom data for the student
    </ResponseField>
    
    <ResponseField name="piar_data" type="object">
      PIAR (Individual Plan for Reasonable Adjustments) data for students with special educational needs
    </ResponseField>
    
    <ResponseField name="guardians" type="array">
      Array of guardian objects associated with this student
      
      <Expandable title="Guardian Object">
        <ResponseField name="id" type="uuid">
          Guardian's unique identifier
        </ResponseField>
        
        <ResponseField name="document_type" type="string">
          Type of identification document
        </ResponseField>
        
        <ResponseField name="document_number" type="string">
          Identification document number
        </ResponseField>
        
        <ResponseField name="full_name" type="string">
          Guardian's full name
        </ResponseField>
        
        <ResponseField name="phone" type="string" optional>
          Contact phone number
        </ResponseField>
        
        <ResponseField name="email" type="string" optional>
          Contact email address
        </ResponseField>
        
        <ResponseField name="address" type="string" optional>
          Physical address
        </ResponseField>
        
        <ResponseField name="occupation" type="string" optional>
          Guardian's occupation
        </ResponseField>
        
        <ResponseField name="workplace" type="string" optional>
          Guardian's workplace
        </ResponseField>
        
        <ResponseField name="is_active" type="boolean">
          Whether the guardian record is active
        </ResponseField>
        
        <ResponseField name="relationship" type="string">
          Relationship to the student (e.g., "Father", "Mother", "Uncle")
        </ResponseField>
        
        <ResponseField name="is_primary" type="boolean">
          Whether this is the primary guardian
        </ResponseField>
        
        <ResponseField name="priority" type="integer">
          Contact priority order (minimum: 1)
        </ResponseField>
        
        <ResponseField name="can_pickup" type="boolean">
          Whether this guardian is authorized to pick up the student
        </ResponseField>
        
        <ResponseField name="is_emergency_contact" type="boolean">
          Whether this guardian should be contacted in emergencies
        </ResponseField>
        
        <ResponseField name="link_id" type="uuid">
          Unique identifier for the student-guardian relationship
        </ResponseField>
        
        <ResponseField name="created_at" type="datetime">
          Timestamp when the relationship was created
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="created_at" type="datetime">
      Timestamp when the student was created
    </ResponseField>
    
    <ResponseField name="updated_at" type="datetime">
      Timestamp when the student was last updated
    </ResponseField>
  </Expandable>
</ResponseField>

## Permissions Required

One of the following permissions:
- `read:students`
- `read:own_students`
- `read:all`

<RequestExample>

```bash cURL
curl -X GET "https://api.athena-erp.com/students?page=1&page_size=20&search=Juan&is_active=true" \
  -H "Authorization: Bearer YOUR_API_TOKEN"
```

```python Python
import requests

response = requests.get(
    "https://api.athena-erp.com/students",
    headers={"Authorization": "Bearer YOUR_API_TOKEN"},
    params={
        "page": 1,
        "page_size": 20,
        "search": "Juan",
        "is_active": True
    }
)

data = response.json()
print(f"Total students: {data['total']}")
```

```javascript JavaScript
const response = await fetch(
  'https://api.athena-erp.com/students?page=1&page_size=20&search=Juan&is_active=true',
  {
    headers: {
      'Authorization': 'Bearer YOUR_API_TOKEN'
    }
  }
);

const data = await response.json();
console.log(`Total students: ${data.total}`);
```
